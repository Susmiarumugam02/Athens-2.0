from rest_framework.decorators import action
from rest_framework.response import Response
from rest_framework import status

class ProjectViewSet(viewsets.ReadOnlyModelViewSet):
    """
    List projects for master admin's tenant
    """
    permission_classes = [IsSuperAdmin | IsMasterAdmin]
    serializer_class = ProjectSerializer
    
    def get_queryset(self):
        user = self.request.user
        
        # Superadmin sees all projects
        if hasattr(user, 'is_superadmin') and user.is_superadmin:
            return Project.objects.all()
        
        # Master admin sees only their tenant's projects
        if hasattr(user, 'masteradmin'):
            return Project.objects.filter(
                tenant=user.masteradmin.tenant
            )
        
        return Project.objects.none()
```

class MasterAdminViewSet(viewsets.ModelViewSet):
    
    @action(detail=False, methods=['post'])
    def create_project_admin(self, request):
        """
        Create a project admin (client/epc/contractor)
        
        Payload:
        {
            "project_id": 1,
            "admin_type": "client",
            "username": "client_admin",
            "company_name": "ABC Corp",
            "registered_address": "123 Main St"
        }
        """
        project_id = request.data.get('project_id')
        admin_type = request.data.get('admin_type')
        username = request.data.get('username')
        company_name = request.data.get('company_name')
        registered_address = request.data.get('registered_address')
        
        # Validate required fields
        if not all([project_id, admin_type, username, company_name, registered_address]):
            return Response(
                {"error": "All fields are required"},
                status=status.HTTP_400_BAD_REQUEST
            )
        
        # Validate project exists and belongs to tenant
        try:
            project = Project.objects.get(id=project_id)
        except Project.DoesNotExist:
            return Response(
                {"error": "Project not found"},
                status=status.HTTP_404_NOT_FOUND
            )
        
        # Check if username already exists
        if MasterAdmin.objects.filter(username=username).exists():
            return Response(
                {"error": "Username already exists"},
                status=status.HTTP_400_BAD_REQUEST
            )
        
        # Generate secure password (16 chars)
        import random
        import string
        password = ''.join(random.choices(
            string.ascii_letters + string.digits + '!@#$%^&*',
            k=16
        )).strip()
        
        # Create admin
        admin = MasterAdmin.objects.create(
            tenant=project.tenant,
            project=project,
            username=username,
            admin_type=admin_type,
            company_name=company_name,
            registered_address=registered_address,
            is_autogenerated_password=True,
            is_password_reset_required=True,
        )
        admin.set_password(password)
        admin.save()
        
        # Return password (ONLY ONCE)
        return Response({
            "username": admin.username,
            "password": password,  # ← CRITICAL: Only shown once
            "admin_type": admin.admin_type,
            "company_name": admin.company_name,
            "registered_address": admin.registered_address,
        }, status=status.HTTP_201_CREATED)
```

class ProjectViewSet(viewsets.ReadOnlyModelViewSet):
    
    @action(detail=True, methods=['get'])
    def admins(self, request, pk=None):
        """
        Get all admins for a project, grouped by type
        
        Response:
        {
            "client": {...},
            "epc": {...},
            "contractors": [...]
        }
        """
        project = self.get_object()
        
        admins = MasterAdmin.objects.filter(project=project)
        
        client_admin = admins.filter(admin_type='client').first()
        epc_admin = admins.filter(admin_type='epc').first()
        contractor_admins = admins.filter(admin_type='contractor')
        
        return Response({
            "client": MasterAdminSerializer(client_admin).data if client_admin else None,
            "epc": MasterAdminSerializer(epc_admin).data if epc_admin else None,
            "contractors": MasterAdminSerializer(contractor_admins, many=True).data,
        })
```

import { useState, useEffect } from 'react';
import { Modal, Form, Input, Select, message } from 'antd';
import { masterAdminService } from '../services/masterAdminService';

interface CreateAdminModalProps {
  open: boolean;
  onClose: () => void;
  onSuccess: () => void;
}

export const CreateAdminModal: React.FC<CreateAdminModalProps> = ({
  open,
  onClose,
  onSuccess,
}) => {
  const [form] = Form.useForm();
  const [loading, setLoading] = useState(false);
  const [projects, setProjects] = useState([]);

  useEffect(() => {
    if (open) {
      fetchProjects();
    }
  }, [open]);

  const fetchProjects = async () => {
    try {
      const response = await masterAdminService.getProjects();
      setProjects(response.data);
    } catch (error) {
      message.error('Failed to load projects');
    }
  };

  const handleSubmit = async (values: any) => {
    setLoading(true);
    try {
      const response = await masterAdminService.createProjectAdmin(values);
      
      // Download credentials
      downloadCredentials(response.data);
      
      message.success('Admin created successfully. Credentials downloaded.');
      form.resetFields();
      onSuccess();
      onClose();
    } catch (error: any) {
      message.error(error.response?.data?.error || 'Failed to create admin');
    } finally {
      setLoading(false);
    }
  };

  const downloadCredentials = (data: any) => {
    const content = `Admin Type: ${data.admin_type.toUpperCase()}
Username: ${data.username}
Password: ${data.password}
Company Name: ${data.company_name}
Registered Address: ${data.registered_address}
`;

    const blob = new Blob([content], { type: 'text/plain' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `${data.admin_type}_admin_credentials_${data.username}.txt`;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    URL.revokeObjectURL(url);
  };

  return (
    <Modal
      title="Create Project Admin"
      open={open}
      onCancel={onClose}
      onOk={() => form.submit()}
      confirmLoading={loading}
      width={600}
    >
      <Form
        form={form}
        layout="vertical"
        onFinish={handleSubmit}
      >
        <Form.Item
          name="project_id"
          label="Project"
          rules={[{ required: true, message: 'Please select a project' }]}
        >
          <Select
            placeholder="Select project"
            showSearch
            optionFilterProp="children"
          >
            {projects.map((project: any) => (
              <Select.Option key={project.id} value={project.id}>
                {project.name}
              </Select.Option>
            ))}
          </Select>
        </Form.Item>

        <Form.Item
          name="admin_type"
          label="Admin Type"
          rules={[{ required: true, message: 'Please select admin type' }]}
        >
          <Select placeholder="Select admin type">
            <Select.Option value="client">Client Admin</Select.Option>
            <Select.Option value="epc">EPC Admin</Select.Option>
            <Select.Option value="contractor">Contractor Admin</Select.Option>
          </Select>
        </Form.Item>

        <Form.Item
          name="username"
          label="Username"
          rules={[
            { required: true, message: 'Please enter username' },
            { pattern: /^[a-zA-Z0-9_]+$/, message: 'Only letters, numbers, and underscore allowed' }
          ]}
        >
          <Input placeholder="Enter username" />
        </Form.Item>

        <Form.Item
          name="company_name"
          label="Company Name"
          rules={[{ required: true, message: 'Please enter company name' }]}
        >
          <Input placeholder="Enter company name" />
        </Form.Item>

        <Form.Item
          name="registered_address"
          label="Registered Office Address"
          rules={[{ required: true, message: 'Please enter registered address' }]}
        >
          <Input.TextArea
            rows={3}
            placeholder="Enter registered office address"
          />
        </Form.Item>
      </Form>
    </Modal>
  );
};
```
